mkdir /usr/local/java
上传安装包
cd /usr/local/java
tar -zxvf jdk-8u351-linux-x64.tar.gz

vi /etc/profile
文末添加
export JAVA_HOME=/usr/local/java/jdk1.8.0_351
export CLASSPATH=$:CLASSPATH:$JAVA_HOME/lib/
export PATH=$PATH:$JAVA_HOME/bin
使改动生效
source /etc/profile
